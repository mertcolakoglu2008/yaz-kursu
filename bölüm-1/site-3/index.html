HTML ÖĞRENDİĞİM ETİKETLER

        !DOCTYPE html = her html dosyasında bulunması gerekir. HTML türü bir dosya olduğunu belirtir.
        meta = arama motorlarına bir web sitesi hakkında bilgi sağlayan HTML etiketidir
        h1 = başlık etiketidir , başlık yazarken kullanılır.
        p = paragraf etiketidir , paragraf yazarken kullanır.
        styling = elementlere şekil , renk , büyüklük ve genişlik gibi özellikler vermeye yarar.
        formatlama etiketleri ör. b , em , strong , small = elementlere büyütme , italik yazma , küçültme gibi özellikler verir.
        a = link etiketidir yazılara resimlere link bırakmaya yarar.
        img = sayfamıza resim koymaya yarar.
